import pygame
import random

# 1. Setup the Game Window
pygame.init()
WIDTH, HEIGHT = 400, 600
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("WU DONG Running")

# 2. Colors & Variables (Change these to change your "Preference")
WHITE = (255, 255, 255)
PLAYER_COLOR = (0, 150, 255)  # Blue
ENEMY_COLOR = (255, 50, 50)   # Red

player_x = WIDTH // 2
player_y = HEIGHT - 100
player_size = 40
speed = 5  # This is your "Up/Down" speed

obstacle_x = random.randint(0, WIDTH - 40)
obstacle_y = -50

clock = pygame.time.Clock()
running = True

# 3. The Main Game Loop
while running:
    screen.fill(WHITE) # Clear the screen
    
    # Check for inputs (Keyboard)
    for event in pygame.event.get():
        if event.type == pygame.QUIT:
            running = False

    keys = pygame.key.get_pressed()
    
    # --- CONTROL LOGIC ---
    if keys[pygame.K_LEFT] and player_x > 0:
        player_x -= 5
    if keys[pygame.K_RIGHT] and player_x < WIDTH - player_size:
        player_x += 5
    if keys[pygame.K_UP]:
        speed += 0.1  # Go faster!
    if keys[pygame.K_DOWN] and speed > 2:
        speed -= 0.1  # Slow down!

    # --- MOVEMENT LOGIC ---
    # The obstacle moves down based on your speed
    obstacle_y += speed 
    
    # If obstacle goes off screen, reset it to the top
    if obstacle_y > HEIGHT:
        obstacle_y = -50
        obstacle_x = random.randint(0, WIDTH - player_size)

    # --- COLLISION (Game Over) ---
    player_rect = pygame.Rect(player_x, player_y, player_size, player_size)
    enemy_rect = pygame.Rect(obstacle_x, obstacle_y, player_size, player_size)
    
    if player_rect.colliderect(enemy_rect):
        print("Game Over! Your final speed was:", round(speed, 1))
        running = False

    # --- DRAWING ---
    pygame.draw.rect(screen, PLAYER_COLOR, player_rect) # Draw Player
    pygame.draw.rect(screen, ENEMY_COLOR, enemy_rect)   # Draw Obstacle
    
    pygame.display.flip() # Update the display
    clock.tick(60)        # Run at 60 frames per second

pygame.quit()
